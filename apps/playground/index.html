Routing for Alpine.js

Add routing and component rendering to any Alpine.js app with a handful of directives — no build step required.

index.html 
<!-- Show a section when the URL matches -->
<div x-route="/about">
  <h1>About</h1>
</div>

<!-- Render a component inline -->
<div x-component="'Counter'"></div>

<!-- Navigate programmatically -->
<button @click="$router.push('/about')">Go</button>

<!-- Access route params -->
<div x-route="/blog/:id">
  <span x-text="$params.id"></span>
</div>
x-route

Show elements when the URL matches

x-component

Render registered components inline

$router

Navigate with push, replace, back

$params

Access dynamic route parameters
